// Then add this script document.addEventListener('DOMContentLoaded', function() { const importContainer = document.getElementById('aurareach-extension-import'); const importBtn = document.getElementById('import-leads-btn'); // Check if extension is available setTimeout(async () => { try { // Test extension connection await window.extensionData.getLeads(); importContainer.style.display = 'block'; } catch (error) { console.log('Extension not connected:', error.message); // Extension not available, hide the import button } }, 1000); // Handle import button click importBtn.addEventListener('click', async () => { importBtn.disabled = true; importBtn.textContent = '⏳ Importing...'; try { // Get leads from extension const leads = await window.extensionData.getLeads(); if (leads.length === 0) { alert('No leads captured in extension!'); importBtn.disabled = false; importBtn.textContent = '📥 Import Leads'; return; } // Send leads to your backend const response = await fetch('/api/sync-leads', { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ leads: leads, timestamp: new Date().toISOString() }) }); if (response.ok) { const result = await response.json(); alert(`✅ Successfully imported ${result.syncedCount} leads!`); // Clear leads from extension after successful sync await window.extensionData.clearLeads(); // Refresh the page or update UI location.reload(); } else { const error = await response.json(); alert(`❌ Import failed: ${error.detail || 'Unknown error'}`); } } catch (error) { alert(`❌ Error: ${error.message}`); } finally { importBtn.disabled = false; importBtn.textContent = '📥 Import Leads'; } }); });
